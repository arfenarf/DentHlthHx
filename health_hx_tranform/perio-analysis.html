diabetes_qy = """match path = (c1:Concept)<-[:ISA*0..3]-(c2)
where tolower(c1.str) contains 'diabet'
or (tolower(c1.str) contains 'insulin' or tolower(c1.str) contains 'humulin')
with c2 match (p:Patient)--(c2)
return p.patient as patient, c2.str as diab_indications;"""

cardio_qy = """match path = (c1:Disease_or_Syndrome)<-[:ISA*0..3]-(c2)
where tolower(c1.str) contains 'heart' or tolower(c1.str) contains 'cardio'
with c2 match (p:Patient)--(c2)
return p.patient as patient, c2.str as cardio_indications;"""

perio_qy = """match (p:Patient)--(pr:Procedure)
where pr.procedure starts with 'D47'
or pr.procedure starts with 'D48'
return p.patient as patient, pr.procedure as perio_indications"""
